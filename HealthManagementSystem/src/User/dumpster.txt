package User;

public class User {
    private String username;
    private String password;
    private String role;

    public User(String username, String password, String role) {
        this.username = username;
        this.password = password;
        this.role = role;
    }

    // Getter methods for accessing user details
    public String getUsername() {
        return username;
    }

    public String getPassword() {
        return password;
    }

    public String getRole() {
        return role;
    }

    // Setter methods for updating user details
    public void setPassword(String password) {
        this.password = password;
    }

    public void setRole(String role) {
        this.role = role;
    }
}

package User;

public class User {
    private String hospitalID;
    private String password;
    private String role;
    private boolean firstLogin;

    // Constructor to initialize a new User with default password
    public User(String hospitalID, String role) {
        this.hospitalID = hospitalID;
        this.password = "password"; // Default password
        this.role = role;
        this.firstLogin = true; // Flag to track if it's the user's first login
    }

    // Getter for Hospital ID
    public String getHospitalID() {
        return hospitalID;
    }

    // Getter for Role
    public String getRole() {
        return role;
    }

    // Method to validate login credentials
    public boolean validateLogin(String enteredPassword) {
        return this.password.equals(enteredPassword);
    }

    // Method to change password
    public boolean changePassword(String oldPassword, String newPassword) {
        if (this.password.equals(oldPassword)) {
            this.password = newPassword;
            this.firstLogin = false; // After changing password, mark as not first login
            return true;
        }
        return false;
    }

    // Check if it is the user's first login
    public boolean isFirstLogin() {
        return firstLogin;
    }

    // Set the user's role (useful for administrator to assign roles if needed)
    public void setRole(String role) {
        this.role = role;
    }

    @Override
    public String toString() {
        return "User [Hospital ID: " + hospitalID + ", Role: " + role + "]";
    }
}
